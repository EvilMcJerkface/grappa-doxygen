00081     // Increment semaphore. Wakes waiters

00093     // Decrement semaphore. Blocks if decrement would leave semaphore < 0.
00094     void decrement( int64_t decr = 1 ) {
00095       DVLOG(5) << this << "/" << Grappa::impl::global_scheduler.get_current_thread() << ": Ready to decrement " << count_ << " by " << decr;
00096       while( count_ - decr < 0 ) {
00097         DVLOG(5) << this << "/" << Grappa::impl::global_scheduler.get_current_thread() << ": Blocking to decrement " << count_ << " by " << decr;
00098         Grappa::wait( this );
00099         DVLOG(5) << this << "/" << Grappa::impl::global_scheduler.get_current_thread() << ": Woken to decrement " << count_ << " by " << decr;
00100       }
00101       check( count_ - decr ); // ensure this adjustment is possible....
00102       count_ -= decr;
00103     }
00104 
00105     // Try to decrement semaphore without blocking. Returns true if it

00107     bool try_decrement( int64_t decr = 1 ) {
00108       DVLOG(5) << this << "/" << Grappa::impl::global_scheduler.get_current_thread() << ": Trying to decrement " << count_ << " by " << decr;
00109       if( count_ - decr < 0 ) {
00110         DVLOG(5) << this << "/" << Grappa::impl::global_scheduler.get_current_thread() << ": Couldn't decrement " << count_ << " by " << decr;
00111         return false;
00112       }
00113       check( count_ - decr ); // ensure this adjustment is possible....
00114       count_ -= decr;
00115       return true;
00116     }
