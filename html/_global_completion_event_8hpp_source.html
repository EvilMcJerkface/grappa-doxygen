183  void send_completion(Core owner, int64_t dec = 1) {
184  if (owner == mycore()) {
185  VLOG(5) << "complete locally ";
186  this->complete(dec);
187  } else {
188  gce_total_remote_completions++;
189  auto& cm = get_completion_msg(owner);
190  if (cm.waiting_to_send()) {
191  cm.completes_to_send++;
192  DVLOG(5) << "flattening completion to Core[" << owner << "] (currently at " << cm.completes_to_send << ")";
193  } else {
194  CHECK_EQ(cm.completes_to_send, 0) << "why haven't we sent these already? cm(" << &cm << ", is_sent: " << cm.is_sent_ << ", is_enqueued: " << cm.is_enqueued_ << ", dest: " << cm.destination_ << ")";
195  DVLOG(5) << "sending " << dec << " to Core[" << owner << "]";
196  cm->dec = dec;

234  CompletionTarget enroll(int64_t inc = 1) {
235  if (inc == 0) return {mycore()};
236 
237  CHECK_GE(inc, 1);
238  count += inc;
239  DVLOG(5) << "enroll " << inc << " -> " << count << " gce("<<this<<")";
240 
241  // first one to have work here
242  if (count == inc) { // count[0 -> inc]
243  event_in_progress = true; // optimization to save checking in wait()
244  // cancel barrier
245  Core co = impl::call(master_core, [this] {
246  cores_out++;
247  return cores_out;
248  });
249  // first one to cancel barrier should make sure other cores are ready to wait
250  if (co == 1) { // cores_out[0 -> 1]
251  event_in_progress = true;
252  call_on_all_cores([this] {
253  event_in_progress = true;
254  });
255  CHECK(event_in_progress);
256  }
257  // block until cancelled
258  CHECK_GT(count, 0);
259  DVLOG(2) << "gce(" << this << " cores_out: " << co << ", count: " << count << ")";
260  }
261 
262  return {mycore()};
263  }

277  void complete(int64_t dec = 1) {
278  count -= dec;
279  DVLOG(4) << "core " << mycore() << " complete (" << count << ") -- gce(" << this << ") cores_out: " << cores_out;
280 
281  // out of work here
282  if (count == 0) { // count[dec -> 0]
283  // enter cancellable barrier
284  send_heap_message(master_core, [this] {
285  cores_out--;
286  DVLOG(4) << "core entered barrier (cores_out:"<< cores_out <<")";
287 
288  // if all are in
289  if (cores_out == 0) { // cores_out[1 -> 0]
290  CHECK_EQ(count, 0);
291 
292  // first, go reset everyone
293  temporary_waking_cores_out = cores();
294  auto re = this->reenroll_count; // remember if we requested reenroll
295  for (Core c = 0; c < cores(); c++) {
296  send_heap_message(c, [this,re] {
297  CHECK_EQ(count, 0);
298  temporary_waking_cv = cv; // capture current list of waiters
299  reset(); // reset, now anyone else calling `wait` should fall through
300  DVLOG(3) << "reset";
301 
302  // if requested, remind cores that we're re-enrolling at the master core
303  if( re ) {
304  DVLOG(5) << "Setting event_in_progress";
305  event_in_progress = true;
306  }
307 
308  // then, once everyone is reset,
309  send_heap_message(master_core, [this,re] {
310  temporary_waking_cores_out--;
311  if (temporary_waking_cores_out == 0) {
312 
313  // if requested, re-enroll cores
314  if( re ) {
315  DVLOG(5) << "Setting count (" << count << ") to " << re << " with event_in_progress " << event_in_progress;
316  count = re; // expect this many completions
317  cores_out = 1; // remember that the master core has outstanding tasks
318  reenroll_count = re; // remember to re-enroll this many cores/tasks next time
319  }
320 
321  // notify everyone to wake
322  for (Core c = 0; c < cores(); c++) {
323  send_heap_message(c, [this] {
324  DVLOG(3) << "broadcast";
325  broadcast(&temporary_waking_cv); // wake anyone who was waiting here
326  temporary_waking_cv.waiters_ = 0;
327  });
328  }
329  }
330  });
331  });
332  }
333  }
334  });
335  }
336  }

340  inline void complete(CompletionTarget ct, int64_t decr = 1) {
341  DVLOG(5) << "called remote complete";
342  if (FLAGS_flatten_completions) {
343  send_completion(ct.core, decr);
344  } else {
345  if (ct.core == mycore()) {
346  complete(decr);
347  } else {
348  if (decr == 1) {
349  // (common case) don't send full 8 bytes just to decrement by 1
350  send_heap_message(ct.core, [this] {
351  complete();
352  });
353  } else {
354  send_heap_message(ct.core, [this,decr] {
355  complete(decr);
356  });
357  }
358  }
359  }
