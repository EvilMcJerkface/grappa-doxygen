118  DVLOG(5) << "complete CompletionEvent";
119  if (ce.core() == mycore()) {

122  ce_remote_completions += decr;
123  if (decr == 1) {
124  // (common case) don't send full 8 bytes just to decrement by 1
125  send_heap_message(ce.core(), [ce] {

128  } else {
129  send_heap_message(ce.core(), [ce,decr] {
