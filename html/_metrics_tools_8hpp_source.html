00059 class RunningStandardDeviation {
00060   private:
00061     double s0;
00062     double s1;
00063     double s2;
00064 
00065   public:
00067     RunningStandardDeviation() {
00068       reset();
00069     }
00070 
00072     void reset() {
00073       s0 = 0.0;
00074       s1 = 0.0;
00075       s2 = 0.0;
00076     }
00077 
00079     void addSample( uint64_t value ) {
00080       s0 += 1;
00081       s1 += value;
00082       s2 += pow(value, 2);
00083     }
00084 
00086     double value() {
00087       double sq = s0*s2 - pow(s1, 2);
00088       if (sq > 0 && s0 > 0) {
00089         return sqrt(sq) / s0;
00090       } else {
00091         return 0;
00092       }
00093     }
00094 
00097     void merge( RunningStandardDeviation other ) {
00098       s0 += other.s0;
00099       s1 += other.s1;
00100       s2 += other.s2;
00101     }
00102 };
