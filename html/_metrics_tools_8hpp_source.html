59 class RunningStandardDeviation {
60  private:

64 
65  public:
67  RunningStandardDeviation() {
68  reset();
69  }

72  void reset() {
73  s0 = 0.0;
74  s1 = 0.0;
75  s2 = 0.0;
76  }

79  void addSample( uint64_t value ) {
80  s0 += 1;
81  s1 += value;
82  s2 += pow(value, 2);
83  }
84 
86  double value() {
87  double sq = s0*s2 - pow(s1, 2);
88  if (sq > 0 && s0 > 0) {
89  return sqrt(sq) / s0;
90  } else {
91  return 0;
92  }
93  }
94 
97  void merge( RunningStandardDeviation other ) {
98  s0 += other.s0;
99  s1 += other.s1;
100  s2 += other.s2;
101  }
102 };
