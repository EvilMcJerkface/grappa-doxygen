00046 template <typename T, T (*ReduceOp)(const T&, const T&)>
00047 class AllReducer {
00048   private:
00049     T * const localSum;
00050     const T * const init;
00051     bool finished;
00052 
00053   public:
00054     AllReducer(T initV) : init(new T(initV))
00055                        , localSum(new T) {}
00056     ~AllReducer() {
00057       delete init;
00058       delete localSum;
00059     }
00060 
00061     void reset() {
00062       finished = false;
00063       *localSum = *init;
00064     }
00065 
00066     void accumulate(T val) {
00067       *localSum = ReduceOp(*localSum, val);
00068     } 
00069 
00076     T finish() {
00077       if (!finished) {
00078         finished = true;
00079         //TODO init version and specialized version for non-template allowed
00080         *localSum = Grappa::allreduce<T,ReduceOp> (*localSum);
00081       }
00082       return *localSum;
00083     }
00084 
00085 
00086     //TODO, a non collective call to finish would be nice
00087     //any such scheme will require knowing where each participant's
00088     //value lives (e.g. process-global variable)
00089 } GRAPPA_BLOCK_ALIGNED;

00092 T reduction(T init, CF f) {
00093   Core master = Grappa::mycore();
00094   auto r = Grappa::symmetric_global_alloc<AllReducerType>();
00095   Grappa::on_all_cores( [=] {
00096       // call the constructor 
00097       new (r.localize()) AllReducerType(init);
00098       r->reset();
00099   });
00100   
00101   // user code, calling accumulate
00102   f(r);
00103 
00104   T result;
00105   Grappa::on_all_cores( [master,r,&result] {
00106     T localcopy = r->finish();
00107     if (Grappa::mycore() == master)

00109   });
00110 
00111   // FIXME: memory leak on r; but free is problematic with symmetric_global_alloc
