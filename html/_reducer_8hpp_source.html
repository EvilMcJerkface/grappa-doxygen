47 class AllReducer {

53  public:

55  , localSum(new T) {}

59  }

61  void reset() {
62  finished = false;
63  *localSum = *init;
64  }

66  void accumulate(T val) {
67  *localSum = ReduceOp(*localSum, val);
68  }

76  T finish() {
77  if (!finished) {
78  finished = true;
79  //TODO init version and specialized version for non-template allowed
80  *localSum = Grappa::allreduce<T,ReduceOp> (*localSum);
81  }
82  return *localSum;
83  }

91 template <typename T, typename AllReducerType, typename CF>
92 T reduction(T init, CF f) {
93  Core master = Grappa::mycore();
94  auto r = Grappa::symmetric_global_alloc<AllReducerType>();
95  Grappa::on_all_cores( [=] {
96  // call the constructor
97  new (r.localize()) AllReducerType(init);
98  r->reset();
99  });
100 
101  // user code, calling accumulate
102  f(r);
103 
104  T result;
105  Grappa::on_all_cores( [master,r,&result] {
106  T localcopy = r->finish();
107  if (Grappa::mycore() == master)
108  result = localcopy;
109  });
110 
111  // FIXME: memory leak on r; but free is problematic with symmetric_global_alloc
112  // Grappa::global_free(r);
113 
114  return result;
115 }
