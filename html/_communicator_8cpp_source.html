00263 size_t Communicator::adjust_footprint(size_t target) {
00264   
00265   auto& sends = FLAGS_log2_concurrent_sends;
00266   auto& recvs = FLAGS_log2_concurrent_receives;
00267   auto& size  =  FLAGS_log2_buffer_size;
00268   
00269   if (estimate_footprint() > target) {
00270     if (mycore == 0) LOG(WARNING) << "Adjusting to fit in target footprint: " << target << " bytes";
00271     while (estimate_footprint() > target) {
00272       // first try adjusting number of concurrent buffers
00273       if (sends > MIN_CONCURRENT_BUFFERS && sends >= recvs) sends--;
00274       else if (recvs > MIN_CONCURRENT_BUFFERS && recvs >= sends) recvs--;
00275       // next try making buffers smaller
00276       else if (size > MIN_LOG2_BUFFER_SIZE) size--;
00277       // otherwise we just have to return what we've got (we're at the minimum allowable footprint)
00278       else break;
00279     }
00280   }
00281   
00282   if (mycore == 0) VLOG(2) << "\nAdjusted:"
00283     << "\n  estimated footprint:      " << estimate_footprint()
00284     << "\n  log2_concurrent_sends:    " << sends
00285     << "\n  log2_concurrent_receives: " << recvs
00286     << "\n  log2_buffer_size:         " << size;
00287   return estimate_footprint();
