61  // Design note: We chose not to use template specialization
62  // on SimpleMetric because enough methods would have to be specialized that it isn't worth it
63  class StringMetric : public impl::MetricBase {
64  public:
65  enum { max_string_size = 2048 };
66 
67  // utility for safe writing using max_string_size
68  static void write_chars(char * dst, std::string newstr, std::string name="(anonymous)") {
69  auto written = snprintf(dst, max_string_size, "%s", newstr.c_str());
70  CHECK( written < max_string_size ) << "StringMetric " << name << " got assigned longer than max " << max_string_size;
71  CHECK( written >= 0 ) << "StringMetric " << name << " assignment failure (code=" << written << ")";
72  }
73 
74  private:

76  write_chars(this->value_, newstr, this->name);
77  }
