00061   // Design note: We chose not to use template specialization 
00062   // on SimpleMetric because enough methods would have to be specialized that it isn't worth it
00063   class StringMetric : public impl::MetricBase {
00064   public:
00065     enum { max_string_size = 2048 };
00066 
00067     // utility for safe writing using max_string_size
00068     static void write_chars(char * dst, std::string newstr, std::string name="(anonymous)") {
00069       auto written = snprintf(dst, max_string_size, "%s", newstr.c_str());
00070       CHECK( written < max_string_size ) << "StringMetric " << name << " got assigned longer than max " << max_string_size;
00071       CHECK( written >= 0 ) << "StringMetric " << name << " assignment failure (code=" << written << ")";
00072     }
00073 
00074   private:
00075     void write_value(std::string newstr) {
00076       write_chars(this->value_, newstr, this->name);
00077     }
00078 
00079   protected:
00080     typedef std::function<std::string(void)> InitFn;
00081     std::string initial_value;
00082     char value_[max_string_size]; // store as a fixed size cstr so we can move it around
00083     InitFn initf_;
00084     
00085 #ifdef VTRACE_SAMPLED
00086     unsigned vt_counter;
00087     static const int vt_type;
00088     
00089     void vt_sample() const;
00090 #endif
00091     
00092   public:
00093     StringMetric(const char * name, std::string initial_value, bool reg_new = true):
00094         initial_value(initial_value), initf_(NULL), impl::MetricBase(name, reg_new) {
00095           write_value(initial_value);
00096 #ifdef VTRACE_SAMPLED
00097         if (StringMetric::vt_type == -1) {
00098           LOG(ERROR) << "warning: VTrace sampling unsupported for this type of StringMetric.";
00099         } else {
00100           vt_counter = VT_COUNT_DEF(name, name, StringMetric::vt_type, VT_COUNT_DEFGROUP);
00101         }
00102 #endif
00103     }
00104    
00105     
00106     virtual std::ostream& json(std::ostream& o) const {
00107       o << '"' << name << "\": " << value_;

00109     }
00110     
00111     virtual void reset() {
00112       if (initf_ != NULL) {
00113         write_value(initf_());

00115         write_value(initial_value);

00117     }

00136     }
