105  void sync() {
106  CompletionEvent ce(keys_to_insert.size()+lookups.size());
107  auto cea = make_global(&ce);
108 
109  for (auto& k : keys_to_insert) {
110  ++hashset_insert_msgs;
111  auto cell = owner->base+owner->computeIndex(k);
112  send_heap_message(cell.core(), [cell,k,cea]{
113  Cell * c = cell.localize();
114  bool found = false;
115  for (auto& e : c->entries) if (e.key == k) { found = true; break; }
116  if (!found) c->entries.emplace_back(k);
117  complete(cea);
118  });
119  }
120  for (auto& e : lookups) { auto k = e.first;
121  ++hashset_lookup_msgs;
122  auto re = e.second;
123  DVLOG(3) << "lookup " << k << " with re = " << re;
124  auto cell = owner->base+owner->computeIndex(k);
125 
126  send_heap_message(cell.core(), [cell,k,cea,re]{
127  Cell * c = cell.localize();
128  bool found = false;
129  for (auto& e : c->entries) if (e.key == k) { found = true; break; }
130 
131  send_heap_message(cea.core(), [cea,re,found]{
132  ResultEntry * r = re;
133  while (r != nullptr) {
134  r->result = found;
135  r = r->next;
136  }
137  complete(cea);
138  });
139  });
140  }
141  ce.wait();
142  }
143  };

215  // Inserts the key if not already in the set
216  //
217  // returns true if the set already contains the key
218  //
219  // synchronous operation
220  void insert( K key ) {
221  ++hashset_insert_ops;
222  if (FLAGS_flat_combining) {
223  proxy.combine([key](Proxy& p){ p.insert(key); return FCStatus::BLOCKED; });
224  } else {
225  ++hashset_insert_msgs;
226  delegate::call(base+computeIndex(key), [key](Cell * c) {
227  // find matching key in the list, if found, no insert necessary
228  for (auto& e : c->entries) if (e.key == key) return;
229 
230  // this is the first time the key has been seen so add it to the list
231  c->entries.emplace_back( key );
232  return;
233  });
234  }

245  void insert_async( K key, F sync) {
246  ++hashset_insert_ops;
247  proxy->insert(key);
248  if (proxy->is_full()) {
249  ++hashset_insert_msgs;
250  spawn([this,sync]{
