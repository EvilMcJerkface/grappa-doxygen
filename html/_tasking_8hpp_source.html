177  void spawn_worker( TF && tf ) {

180  Worker * th = impl::worker_spawn( Grappa::impl::global_scheduler.get_current_thread(), &Grappa::impl::global_scheduler,

197 void run(FP fp) {
198 #ifdef GRAPPA_TRACE
199  TAU_PROFILE("run_user_main()", "(user code entry)", TAU_USER);
200 #endif
201 #ifdef VTRACE
202  VT_TRACER("run_user_main()");
203 #endif
204 #ifdef VTRACE_SAMPLED
205  // this doesn't really add anything to the profiled trace
206  //Grappa_Grappa::impl::take_tracing_sample();
207 #endif
208 
209  if( global_communicator.mycore == 0 ) {
210  CHECK_EQ( Grappa::impl::global_scheduler.get_current_thread(), master_thread ); // this should only be run at the toplevel
211 
212  // create user_main as a private task
213  //Grappa_privateTask( &user_main_wrapper<A>, fp, args );
214  Grappa::spawn( [&fp] {
215  Grappa_global_queue_initialize();
216  fp();
217  Metrics::dump_stats_blob();
218  Grappa_end_tasks();
219  } );
220 
221  DVLOG(5) << "Spawned user_main";
222 
223  // spawn 1 extra worker that will take user_main
224  Grappa::impl::global_scheduler.createWorkers( 1 );
225  }
226 
227  // spawn starting number of worker coroutines
228  Grappa::impl::global_scheduler.createWorkers( FLAGS_num_starting_workers );
229  Grappa::impl::global_scheduler.allow_active_workers(-1); // allow all workers to be active
230 
231  StateTimer::init();
232 
233  // bypass atexit() handlers. TODO: figure out what causes crashes when exit()ing inside grappa threads
234  on_exit( [] ( int retval, void * payload) { _exit(retval); }, nullptr );
235 
236  // start the scheduler
237  Grappa::impl::global_scheduler.run( );
238 
239 #ifdef VTRACE_SAMPLED
240  // this doesn't really add anything to the profiled trace
241  //Grappa_Grappa::impl::take_tracing_sample();
242 #endif
243 }
