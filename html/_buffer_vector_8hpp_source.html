00042 /* Vector that exposes its storage array as read-only.
00043  * increases in size as elements are added.
00044  *
00045  * Only supports insertions. Insertions allowed only
00046  * while vector is in write only (WO) mode and 
00047  * getting the buffer pointer only allowed in read only mode
00048  */
00049 
00050 template < typename T >
00051 class BufferVector {
00052 
00053   private:
00054     T * buf;
00055     size_t size;
00056     uint64_t nextIndex;
00057 
00058     enum OpMode { RO, WO };
00059     OpMode mode;
00060 
00061   public:
00062     BufferVector( size_t capacity = 2 ) 
00063       : buf ( static_cast<T*>(Grappa::impl::locale_shared_memory.allocate( capacity*sizeof(T) )) )   // new T[capacity]
00064       , mode( WO )
00065       , nextIndex( 0 )
00066       , size( capacity ) { }
00067 
00068     ~BufferVector() {
00069       Grappa::impl::locale_shared_memory.deallocate( buf );
00070     }
00071 
00072     void setWriteMode() {
00073       CHECK( mode != WO ) << "already in WO mode";
00074       mode = WO;
00075     }
00076 
00077     void setReadMode() {
00078       CHECK( mode != RO ) << "already in RO mode";
00079       mode = RO;
00080     }
00081 
00082     void insert( const T& v ) {
00083       CHECK( mode == WO ) << "Must be in WO mode to insert";
00084 
00085       if ( nextIndex == size ) {
00086         // expand the size of the buffer
00087         size_t newsize = size * 2;
00088         VLOG(3) << "New size of cell size: " << newsize << " bytes: " << newsize*sizeof(T);
00089         T * newbuf = static_cast<T*>(Grappa::impl::locale_shared_memory.allocate( newsize*sizeof(T) ));      // new T[newsize]
00090         memcpy( newbuf, buf, size*sizeof(T) );
00091         Grappa::impl::locale_shared_memory.deallocate( buf );   // delete buf
00092         size = newsize;
00093         buf = newbuf;
00094       }
00095 
00096       // store element and increment index
00097       buf[nextIndex++] = v;
00098     }
00099 
00100     GlobalAddress<const T> getReadBuffer() const {
00101       CHECK( mode == RO ) << "Must be in RO mode to see buffer";
00102       return make_global( const_cast<const T*>(buf) );
00103     }
00104 
00105     size_t getLength() const {
00106       return nextIndex; 
00107     }
00108 
00109     /* to close the safety loop, RO clients should release buffers,
00110      * allowing setWriteMode() to check that all buffers are released
00111      * but currently we will avoid this extra communication
00112      *
00113      * If we want more fine-grained concurrent R/W then it is better
00114      * not to expose this getBuffer type of interface 
00115      */

00117     template < typename T1 >
00118     friend std::ostream& operator<<( std::ostream& o, const BufferVector<T1>& v );
00119 };
