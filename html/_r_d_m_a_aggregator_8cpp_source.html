00205     // (assume all locales have same core count)

00233         dest_core_for_locale_[i] = -1;
00234       } else {
00235         // guess at correct destination core on the node
00236         Core offset = Grappa::mylocale() / locales_per_core;
00237         if( offset >= Grappa::locale_cores() ) { // if we guess too high
00238           // give it to the core that would have been responsible for

01020   class MessageListChooser {
01021   private:
01022     // indices of source core message lists
01023     Core dest_cores_start_;
01024     Core dest_cores_end_;
01025     
01026     // cores in this locale to pull from
01027     Core source_cores_start_;
01028     Core source_cores_end_;
01029     
01030     // where are we right now?
01031     Core current_dest_core_;
01032     Core current_source_core_;
01033     
01034     bool done;
01035     
01036   public:
01037     MessageListChooser() = delete;
01038     MessageListChooser( const MessageListChooser& ) = delete;
01039     MessageListChooser( MessageListChooser&& ) = delete;
01040 
01041     MessageListChooser( Core dest_start, Core dest_end, Core source_start, Core source_end ) 
01042       : dest_cores_start_( dest_start )
01043       , dest_cores_end_( dest_end )
01044       , source_cores_start_( source_start )
01045       , source_cores_end_( source_end )
01046       , current_dest_core_( dest_cores_start_ )
01047       , current_source_core_( source_cores_start_ )
01048       , done( false )
01049     { 
01050       DVLOG(4) << __func__ << "Initialized with dest range " << dest_cores_start_ << "/" << dest_cores_end_
01051                << " source range " << source_cores_start_ << "/" << source_cores_end_;
01052     }
01053 
01054     CoreData * get_next_list( Core * c ) {
01055       CoreData * cd = NULL;
01056 
01057       if( !done ) {
01058         do {
01059           // load next core data pointer. ideally this will have messages most of the time.
01060           cd = global_rdma_aggregator.coreData( current_dest_core_, current_source_core_ );
01061           *c = current_dest_core_;
01062 
01063           DVLOG(4) << __func__ << ": Fetched CoreData ptr " << cd << " dest core " << current_dest_core_
01064                    << " with cd " << cd << " messageList " << cd->messages_.raw_;
01065 
01066           // got nothing, so go to next source core in this locale
01067           current_source_core_++;
01068           
01069           // if we've checked all the local cores for this destination
01070           if( current_source_core_ >= source_cores_end_ ) {
01071             // reset source
01072             current_source_core_ = source_cores_start_;
01073             // and increment dest
01074             current_dest_core_++;
01075             // if we've also checked all the destinations
01076             if( current_dest_core_ >= dest_cores_end_ ) {
01077               done = true; // we're done.
01078             }
01079           }
01080 
01081           // if we still have something to fetch, prefetch for next
01082           // call (assuming we found something earlier and don't have
01083           // to keep searching....)
01084           if( !done ) {
01085             CoreData * next = global_rdma_aggregator.coreData( current_dest_core_, current_source_core_ );
01086             char * pf = reinterpret_cast< char* >( next );
01087             __builtin_prefetch( pf +  0, 1, prefetch_type );
01088 #ifdef DOUBLE_LINE_PREFETCH
01089             __builtin_prefetch( pf + 64, 1, prefetch_type );
01090 #endif
01091           }
01092         } while( 0 == cd->messages_.raw_ && !done );
01093       }
01094       
01095       return cd;
01096     }
