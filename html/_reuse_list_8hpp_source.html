74  // block until we have a buffer to process
75  T * block_until_pop() {
76  while( empty() ) {
77  DVLOG(5) << __PRETTY_FUNCTION__ << ": blocking";
78  Grappa::wait( &cv );
79  }
80 
81  T * b = list_;
82 
83  // put back rest of list
84  list_ = b->get_next();
85 
86  DVLOG(5) << __PRETTY_FUNCTION__ << ": popping " << b << " (count " << count_ << ") list=" << list_;
87 
88  // make sure the buffer is not part of any list
89  b->set_next( NULL );
90 
91  // record the get
92  count_--;
93 
94  return b;
95  }
96 
98  T * try_pop() {
99  if( !empty() ) {
100  return block_until_pop();
101  } else {
102  return NULL;
103  }
104  }

108
