00067     void process( T t ) {
00068       n++;
00069       double delta = t - mean;
00070       mean += delta / n;
00071       M2 += delta * (t - mean);
00072       if (n == 1) {
00073         min = t;
00074         max = t;
00075       }
00076       if (t > max) max = t;
00077       if (t < min) min = t;
00078     }
00079     
00080     double variance() const {
00081       return (n<2) ? 0.0 : M2 / (n - 1);
00082     }
00083 
00084     double stddev() const {
00085       return sqrt( variance() );
