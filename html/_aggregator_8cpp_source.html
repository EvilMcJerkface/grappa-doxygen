164 void Aggregator::deaggregate( ) {
165  GRAPPA_FUNCTION_PROFILE( GRAPPA_COMM_GROUP );
166 #ifdef VTRACE_FULL
167  VT_TRACER("deaggregate");
168 #endif
169  StateTimer::enterState_deaggregation();
170  while( !received_AM_queue_.empty() ) {
171  DVLOG(5) << "deaggregating";
172  // TODO: too much copying
173  ReceivedAM amp = received_AM_queue_.front();
174 
175 #ifdef STL_DEBUG_ALLOCATOR
176  if( aggregator_access_control_active ) STLMemDebug::BaseAllocator::getMemMgr().setAccessMode(STLMemDebug::memReadWrite);
177 #endif
178  received_AM_queue_.pop();
179 #ifdef STL_DEBUG_ALLOCATOR
180  if( aggregator_access_control_active ) STLMemDebug::BaseAllocator::getMemMgr().setAccessMode(STLMemDebug::memReadOnly);
181 #endif
182 
183  DVLOG(5) << "deaggregating message of size " << amp.size_;
184  uintptr_t msg_base = reinterpret_cast< uintptr_t >( amp.buf_ );
185  for( unsigned int i = 0; i < amp.size_; ) {
186  AggregatorGenericCallHeader * header = reinterpret_cast< AggregatorGenericCallHeader * >( msg_base );
187  AggregatorAMHandler fp = reinterpret_cast< AggregatorAMHandler >( header->function_pointer );
188  void * args = reinterpret_cast< void * >( msg_base +
189  sizeof( AggregatorGenericCallHeader ) );
190  void * payload = reinterpret_cast< void * >( msg_base +
191  sizeof( AggregatorGenericCallHeader ) +
192  header->args_size );
193  if( header->destination == Grappa::mycore() ) { // for us?
194 
195  stats.record_deaggregation( sizeof( AggregatorGenericCallHeader ) + header->args_size + header->payload_size );
196  // trace fine-grain communication
197 #ifdef GRAPPA_TRACE
198  if (FLAGS_record_grappa_events) {
199  // TODO: good candidate for TAU_CONTEXT_EVENT
200  int fn_p_tag = aggregator_trace_tag( fp );
201  TAU_TRACE_RECVMSG(fn_p_tag, header->source, header->args_size + header->payload_size );
202  }
203 #endif
204 
205 #ifdef VTRACE_FULL
206  {
207  VT_RECV( vt_agg_commid_, header->tag, sizeof( AggregatorGenericCallHeader ) + header->args_size + header->payload_size );
208  }
209 #endif
210 
211 
212  DVLOG(5) << "calling " << *header
213  << " with args " << args
214  << " and payload " << payload;
215 
216  {
217  GRAPPA_PROFILE( deag_func_timer, "deaggregate execution", "", GRAPPA_USERAM_GROUP );
218  fp( args, header->args_size, payload, header->payload_size ); // execute
219  }
220  } else { // not for us, so forward towards destination
221  DVLOG(5) << "forwarding " << *header
222  << " with args " << args
223  << " and payload " << payload;
224  stats.record_forward( sizeof( AggregatorGenericCallHeader ) + header->args_size + header->payload_size );
225  aggregate( header->destination, fp, args, header->args_size, payload, header->payload_size );
226  }
227  i += sizeof( AggregatorGenericCallHeader ) + header->args_size + header->payload_size;
228  msg_base += sizeof( AggregatorGenericCallHeader ) + header->args_size + header->payload_size;
229  }
230  }
231 }
