00588 static void write_locked_range( const char * filename, size_t offset, const char * buf, size_t size ) {
00589   int fd;
00590 
00591   PCHECK( (fd = open( filename, O_WRONLY )) >= 0 );
00592 
00593   // lock range in file
00594   struct flock lock;
00595   lock.l_type = F_WRLCK;
00596   lock.l_whence = SEEK_SET;
00597   lock.l_start = offset;
00598   lock.l_len = size;
00599   PCHECK( fcntl( fd, F_SETLK, &lock ) >= 0 ) << "Could not obtain record lock on file";
00600 
00601   // seek to range
00602   int64_t new_offset = 0;
00603   PCHECK( (new_offset = lseek( fd, offset , SEEK_SET )) >= 0 );
00604   CHECK_EQ( new_offset, offset );
00605       
00606   // write
00607   PCHECK( write( fd, buf, size ) >= 0 );
00608 
00609   // release lock on file range
00610   lock.l_type = F_UNLCK;
00611   PCHECK( fcntl( fd, F_SETLK, &lock ) >= 0 ) << "Could not release record lock on file";
00612 
00613   // close file
