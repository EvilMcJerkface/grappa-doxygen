00099     void check_invariants() {
00100       uint64_t max=std::numeric_limits<uint64_t>::min();
00101       uint64_t min=std::numeric_limits<uint64_t>::max();
00102       uint64_t sum=0;
00103       for ( uint64_t i=0; i<num_queues; i++ ) {
00104         max = _max<uint64_t>( max, queues[i].length() );
00105         min = _min<uint64_t>( min, queues[i].length() );
00106         sum += queues[i].length();
00107       }
00108       CHECK( static_cast<int64_t>(max) - static_cast<int64_t>(min) >= 0 );
00109       CHECK( max - min <= 1 ) << "min=" << min << " max=" << max;
00110       CHECK( sum == len );
