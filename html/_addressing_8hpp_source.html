224  intptr_t signextended = (storage_ << pointer_shift_val) >> pointer_shift_val;

239  inline T * localize(Core nid = -1) const {
240  if (is_2D()) return pointer();
241 
242  if (nid == -1) nid = global_communicator.mycore;
243  T * local_base;
244  size_t block_elems = block_size / sizeof(T);
245  T * block_base = block_min().pointer();
246  if (nid < core()) {
247  local_base = block_base+block_elems;
248  } else if (nid > core()) {
