00118   size_t cacheline_count = sz / CACHE_LINE_SIZE;

00120   
00121   if( sz > MAX_POOL_MESSAGE_SIZE ) {
00122     shared_pool_alloc_toobig++;
00123     return locale_alloc_aligned<char>(CACHE_LINE_SIZE, sz);

00125     // record the pool allocation (bucketed by number of cachelines)
