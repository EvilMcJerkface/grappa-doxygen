00081   void do_memcpy_locally(GlobalAddress<T> dst, GlobalAddress<T> src, size_t nelem) {
00082     typedef typename Incoherent<T>::WO Writeback;
00083     auto src_end = src+nelem;
00084     // T * local_base = src.localize(), * local_end = (src+nelem).localize();
00085     int64_t nfirstcore = src.block_max() - src;
00086     int64_t nlastcore = src_end - src_end.block_min();
00087     int64_t nmiddle = nelem - nfirstcore - nlastcore;
00088     
00089     const size_t nblock = block_size / sizeof(T);
00090     
00091     CHECK_EQ(nmiddle % nblock, 0);
00092 
00093     auto src_start = src;
00094     if (src.core() == mycore()) {
00095       int64_t nfirstcore = src.block_max() - src;
00096       if (nfirstcore > 0 && nlastcore != nblock) {
00097         DVLOG(3) << "nfirstcore = " << nfirstcore;
00098         Writeback w(dst, nfirstcore, src.pointer());
00099         src_start += nfirstcore;
00100       }
00101     }
00102     if ((src_end-1).core() == mycore()) {
00103       int64_t nlastcore = src_end - src_end.block_min();
00104       int64_t index = nelem - nlastcore;
00105       if (nlastcore > 0 && nlastcore != nblock) {
00106         DVLOG(3) << "nlastcore = " << nlastcore << ", index = " << index;
00107         CHECK((src+index).core() == mycore());
00108         Writeback w(dst+index, nlastcore, (src+index).pointer());
00109         src_end -= nlastcore;
00110       }
00111     }
00112     
00113     auto * local_base = src_start.localize();
00114     size_t nlocal_trimmed = src_end.localize() - local_base;
00115     CHECK_EQ((nlocal_trimmed) % nblock, 0);
00116     size_t nlocalblocks = nlocal_trimmed/nblock;
00117     Writeback * ws = locale_alloc<Writeback>(nlocalblocks);
00118     for (size_t i=0; i<nlocalblocks; i++) {
00119       size_t j = make_linear(local_base+(i*nblock))-src;
00120       new (ws+i) Writeback(dst+j, nblock, local_base+(i*nblock));
00121       ws[i].start_release();
00122     }
00123     for (size_t i=0; i<nlocalblocks; i++) { ws[i].block_until_released(); }
00124     locale_free(ws);
00125   }
