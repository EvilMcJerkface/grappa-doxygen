80  template< typename T >
81  void do_memcpy_locally(GlobalAddress<T> dst, GlobalAddress<T> src, size_t nelem) {
82  typedef typename Incoherent<T>::WO Writeback;
83  auto src_end = src+nelem;
84  // T * local_base = src.localize(), * local_end = (src+nelem).localize();
85  int64_t nfirstcore = src.block_max() - src;
86  int64_t nlastcore = src_end - src_end.block_min();
87  int64_t nmiddle = nelem - nfirstcore - nlastcore;
88 
89  const size_t nblock = block_size / sizeof(T);
90 
91  CHECK_EQ(nmiddle % nblock, 0);
92 
93  auto src_start = src;
94  if (src.core() == mycore()) {
95  int64_t nfirstcore = src.block_max() - src;
96  if (nfirstcore > 0 && nlastcore != nblock) {
97  DVLOG(3) << "nfirstcore = " << nfirstcore;
98  Writeback w(dst, nfirstcore, src.pointer());
99  src_start += nfirstcore;
100  }
101  }
102  if ((src_end-1).core() == mycore()) {
103  int64_t nlastcore = src_end - src_end.block_min();
104  int64_t index = nelem - nlastcore;
105  if (nlastcore > 0 && nlastcore != nblock) {
106  DVLOG(3) << "nlastcore = " << nlastcore << ", index = " << index;
107  CHECK((src+index).core() == mycore());
108  Writeback w(dst+index, nlastcore, (src+index).pointer());
109  src_end -= nlastcore;
110  }
111  }
112 
113  auto * local_base = src_start.localize();
114  size_t nlocal_trimmed = src_end.localize() - local_base;
115  CHECK_EQ((nlocal_trimmed) % nblock, 0);
116  size_t nlocalblocks = nlocal_trimmed/nblock;
117  Writeback * ws = locale_alloc<Writeback>(nlocalblocks);
118  for (size_t i=0; i<nlocalblocks; i++) {
119  size_t j = make_linear(local_base+(i*nblock))-src;
120  new (ws+i) Writeback(dst+j, nblock, local_base+(i*nblock));
121  ws[i].start_release();
122  }
123  for (size_t i=0; i<nlocalblocks; i++) { ws[i].block_until_released(); }
124  locale_free(ws);
125  }
126 }
