00088     template< TaskMode B,
00089               GlobalCompletionEvent * C = nullptr,
00090               int64_t Threshold = USE_LOOP_THRESHOLD_FLAG,
00091               typename F = decltype(nullptr) >
00092     void loop_decomposition(int64_t start, int64_t iterations, F loop_body) {
00093       DVLOG(4) << "< " << start << " : " << iterations << ">";
00094       
00095       if (iterations == 0) {
00096         return;
00097       } else if ((Threshold == USE_LOOP_THRESHOLD_FLAG && iterations <= FLAGS_loop_threshold)
00098                  || iterations <= Threshold) {
00099         loop_body(start, iterations);
00100         return;
00101       } else {
00102         // spawn right half
00103         int64_t rstart = start+(iterations+1)/2, riters = iterations/2;
00104         Core origin = mycore();
00105         
00106         // pack these 3 into 14 bytes so we still have room for a full 8-byte word from user
00107         // (also need to count 2 bytes from lambda overhead or something)
00108         struct { long rstart:48, riters:48, origin:16; } packed = { rstart, riters, origin };
00109         
00110         if (C) C->enroll();
00111         spawn<B>([packed, loop_body] {
00112           loop_decomposition<B,C,Threshold>(packed.rstart, packed.riters, loop_body);
00113           if (C) C->send_completion(packed.origin);
00114         });
00115         
00116         // left side here
00117         loop_decomposition<B,C,Threshold>(start, (iterations+1)/2, loop_body);
00118       }
00119     }

00132     template<TaskMode B, SyncMode S, GlobalCompletionEvent * C, int64_t Threshold, typename F>
00133     void forall_here(int64_t start, int64_t iters, F loop_body,
00134                      void (F::*mf)(int64_t,int64_t) const)
00135     {
00136       struct HeapF {
00137        const F loop_body;
00138        int64_t refs;
00139        HeapF(F loop_body, int64_t refs): loop_body(std::move(loop_body)), refs(refs) {}
00140        void ref(int64_t delta) {
00141          refs += delta;
00142          if (refs == 0) delete this;
00143        }
00144       };
00145       
00146       if (C == nullptr && S == SyncMode::Blocking) {
00147         if (B == TaskMode::Bound) {
00148           CompletionEvent ce(iters);
00149           impl::loop_decomposition<B,C,Threshold>(start, iters,
00150           [&loop_body,&ce](int64_t s, int64_t n){
00151             loop_body(s, n);
00152             ce.complete(n);
00153           });
00154           ce.wait();
00155         } else {
00156           CHECK(false) << "unimplemented, sorry!";
00157         }
00158       } else if (C && S == SyncMode::Async && B == TaskMode::Bound
00159           && sizeof(F) > 8
00160           && C->get_shared_ptr<F>() == nullptr) {
00161         auto hf = new HeapF(loop_body, iters);
00162         impl::loop_decomposition<B,C,Threshold>(start, iters,
00163             [hf](int64_t s, int64_t n){
00164           hf->loop_body(s, n);
00165           hf->ref(-n);
00166         });
00167       } else {
00168         impl::loop_decomposition<B,C,Threshold>(start, iters, loop_body);
00169         if (S == SyncMode::Blocking && C) C->wait();
00170       }
00171     }

00238     void forall(int64_t start, int64_t iters, F loop_body,
00239         void (F::*mf)(int64_t,int64_t) const) {
00240       static_assert( C != nullptr, "GCE template arg cannot be null; need the GCE to store shared args");
00241     
00242       Core origin = mycore();
00243       C->enroll(cores());
00244       
00245       on_all_cores([start,iters,origin,loop_body]{
00246         
00247         range_t r = blockDist(start, start+iters, mycore(), cores());
00248         
00249         if (sizeof(loop_body) > 8) {
00250           
00251           CHECK(S != SyncMode::Async) << "Cannot do 'async' with a lambda > 8 bytes.";
00252           
00253           // initialize this on all cores before starting any tasks
00254           C->set_shared_ptr(&loop_body);
00255           barrier();
00256           
00257           forall_here<B,SyncMode::Async,C,Threshold>(r.start, r.end-r.start, [](int64_t s, int64_t n){
00258             auto& loop_body = *C->get_shared_ptr<F>();
00259             loop_body(s,n);
00260           });
00261         
00262           C->send_completion(origin);
00263           C->wait();
00264           C->set_shared_ptr(nullptr);
00265           
00266         } else {
00267           
00268           forall_here<B,SyncMode::Async,C,Threshold>(r.start, r.end-r.start, loop_body);
00269           C->send_completion(origin);
00270           if (S == SyncMode::Blocking) C->wait();
00271           
00272         }
00273       });

00339   template< typename T >
00340   std::pair<Core,Core> cores_with_elements(GlobalAddress<T> base, size_t nelem) {
00341     Core start_core = base.core();
00342     Core nc = cores();
00343     Core fc;
00344     int64_t nbytes = nelem*sizeof(T);
00345     auto end = base+nelem;
00346     if (nelem > 0) { fc = 1; }
00347   
00348     size_t block_elems = std::max<size_t>(1, block_size / sizeof(T));
00349     int64_t nfirstcore = base.block_max() - base;
00350     int64_t n = nelem - nfirstcore;
00351     
00352     if (n > 0) {
00353       int64_t nrest = n / block_elems;
00354       if (nrest >= nc-1) {
00355         fc = nc;
00356       } else {
00357         fc += nrest;
00358         if ((end - end.block_min()) && end.core() != base.core()) {
00359           fc += 1;
00360         }
00361       }
00362     }
00363     
00364     return std::make_pair( start_core, fc );
00365   }

00420               GlobalCompletionEvent * GCE, int64_t Threshold,

00422     void forall(GlobalAddress<T> base, int64_t nelems, F loop_body,
00423                 void (F::*mf)(int64_t,T&) const)
00424     {

00426         auto block_elems = block_size / sizeof(T);
00427         auto a = make_linear(first);
00428         auto n_to_boundary = a.block_max() - a;

00431         for (int64_t i=0; i<niters; i++,index++,n_to_boundary--) {
00432           if (n_to_boundary == 0) {
