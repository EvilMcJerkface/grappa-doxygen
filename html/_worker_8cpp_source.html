59 void insert_coro( Worker * c ) {
60  // try to insert us in list going left
61  if( all_coros ) {
62  CHECK( all_coros->tracking_prev == NULL ) << "Head coroutine should not have a prev";
63  all_coros->tracking_prev = c;
64  }
65 
66  // insert us in list going right
67  c->tracking_next = all_coros;
68  all_coros = c;

73 void remove_coro( Worker * c ) {
74  // is there something to our left?

79  // is there something to our right?
80  if( c->tracking_next ) {
81  // remove us from next list
82  c->tracking_next->tracking_prev = c->tracking_prev;
83  }

135 void coro_spawn(Worker * me, Worker * c, coro_func f, size_t ssize) {
136  CHECK(c != NULL) << "Must provide a valid Worker";
137  c->running = 0;
138  c->suspended = 0;
139  c->idle = 0;
140 
141  // allocate stack and guard page
142  c->base = Grappa::impl::locale_shared_memory.allocate_aligned( ssize+4096*2, 4096 );
143  CHECK_NOTNULL( c->base );
144  c->ssize = ssize;
145 
146  // set stack pointer
147  c->stack = (char*) c->base + ssize + 4096 - current_stack_offset;
148 
149  // try to make sure we don't stuff all our stacks at the same cache index
150  const int num_offsets = 128;
151  const int cache_line_size = 64;
152  current_stack_offset += FLAGS_stack_offset;
153  current_stack_offset &= ((cache_line_size * num_offsets) - 1);
154 
155  c->tracking_prev = NULL;
156  c->tracking_next = NULL;
157 
158 #ifdef ENABLE_VALGRIND
159  c->valgrind_stack_id = VALGRIND_STACK_REGISTER( (char *) c->base + 4096, c->stack );
160 #endif
161 
162  // clear stack
163  memset(c->base, 0, ssize+4096*2);
164 
165 #ifdef GUARD_PAGES_ON_STACK
166  // arm guard page
167  checked_mprotect( c->base, 4096, PROT_NONE );
168  checked_mprotect( (char*)c->base + ssize + 4096, 4096, PROT_NONE );
169 #endif
170 
171  // set up coroutine to be able to run next time we're switched in
172  makestack(&me->stack, &c->stack, f, c);
173 
174  insert_coro( c ); // insert into debugging list of coros
175 
176 #ifdef CORO_PROTECT_UNUSED_STACK
177  // disable writes to stack until we're swtiched in again.
178  checked_mprotect( (void*)((intptr_t)c->base + 4096), ssize, PROT_READ );
179  checked_mprotect( (void*)(c), 4096, PROT_READ );
180 #endif
181 
182  total_coros++;
183 }

212 void destroy_coro(Worker * c) {
213  total_coros--;
214 #ifdef ENABLE_VALGRIND
215  if( c->valgrind_stack_id != -1 ) {
216  VALGRIND_STACK_DEREGISTER( c->valgrind_stack_id );
217  }
218 #endif
219  if( c->base != NULL ) {
220  // disarm guard page
221  checked_mprotect( c->base, 4096, PROT_READ | PROT_WRITE );
222  checked_mprotect( (char*)c->base + c->ssize + 4096, 4096, PROT_READ | PROT_WRITE );
223 #ifdef CORO_PROTECT_UNUSED_STACK
224  // enable writes to stack so we can deallocate
225  checked_mprotect( (void*)((intptr_t)c->base + 4096), c->ssize, PROT_READ | PROT_WRITE );
226  checked_mprotect( (void*)(c), 4096, PROT_READ | PROT_WRITE );
227 #endif
228  remove_coro(c); // remove from debugging list of coros
229  Grappa::impl::locale_shared_memory.deallocate(c->base);
230  }
