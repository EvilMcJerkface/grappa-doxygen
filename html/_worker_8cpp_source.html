00060   // try to insert us in list going left
00061   if( all_coros ) {

00135 void coro_spawn(Worker * me, Worker * c, coro_func f, size_t ssize) {
00136   CHECK(c != NULL) << "Must provide a valid Worker";
00137   c->running = 0;
00138   c->suspended = 0;
00139   c->idle = 0;
00140 
00141   // allocate stack and guard page
00142   c->base = Grappa::impl::locale_shared_memory.allocate_aligned( ssize+4096*2, 4096 );
00143   CHECK_NOTNULL( c->base );
00144   c->ssize = ssize;
00145 
00146   // set stack pointer
00147   c->stack = (char*) c->base + ssize + 4096 - current_stack_offset;
00148 
00149   // try to make sure we don't stuff all our stacks at the same cache index
00150   const int num_offsets = 128;
00151   const int cache_line_size = 64;
00152   current_stack_offset += FLAGS_stack_offset;
00153   current_stack_offset &= ((cache_line_size * num_offsets) - 1);
00154   
00155   c->tracking_prev = NULL;
00156   c->tracking_next = NULL;
00157 
00158 #ifdef ENABLE_VALGRIND
00159   c->valgrind_stack_id = VALGRIND_STACK_REGISTER( (char *) c->base + 4096, c->stack );
00160 #endif
00161 
00162   // clear stack
00163   memset(c->base, 0, ssize+4096*2);
00164 
00165 #ifdef GUARD_PAGES_ON_STACK
00166   // arm guard page
00167   checked_mprotect( c->base, 4096, PROT_NONE );
00168   checked_mprotect( (char*)c->base + ssize + 4096, 4096, PROT_NONE );
00169 #endif
00170 
00171   // set up coroutine to be able to run next time we're switched in
00172   makestack(&me->stack, &c->stack, f, c);
00173   
00174   insert_coro( c ); // insert into debugging list of coros
00175 
00176 #ifdef CORO_PROTECT_UNUSED_STACK
00177   // disable writes to stack until we're swtiched in again.
00178   checked_mprotect( (void*)((intptr_t)c->base + 4096), ssize, PROT_READ );
00179   checked_mprotect( (void*)(c), 4096, PROT_READ );
00180 #endif
00181 
00182   total_coros++;
00183 }
