00125 size_t TaskManager::adjust_footprint(size_t target) {
00126   if (estimate_footprint() > target) {
00127     MASTER_ONLY LOG(WARNING) << "Adjusting to fit in target footprint: " << target << " bytes";
00128     while (estimate_footprint() > target) {
00129       // first try making the steal queue smaller
00130       if (steal_queue_size > 4) steal_queue_size /= 2;
00131       // otherwise we have to start removing workers
00132       else if (FLAGS_num_starting_workers > 4) FLAGS_num_starting_workers--;
00133     }
00134     MASTER_ONLY VLOG(2) << "\nAdjusted:"
00135       << "\n  estimated footprint:  " << estimate_footprint()
00136       << "\n  steal_queue_size:     " << steal_queue_size
00137       << "\n  num_starting_workers: " << FLAGS_num_starting_workers;
00138   }
00139   return estimate_footprint();
00140 }

00143 void TaskManager::init ( Core localId_arg, Core * neighbors_arg, Core numLocalNodes_arg ) {
00144   if ( FLAGS_load_balance.compare(        "none" ) == 0 ) {
00145     doSteal = false; doShare = false; doGQ = false;
00146   } else if ( FLAGS_load_balance.compare( "steal" ) == 0 ) {
00147     doSteal = true; doShare = false; doGQ = false;
00148   } else if ( FLAGS_load_balance.compare( "share" ) == 0 ) {
00149     CHECK( false ) << "--load_balance=share currently unsupported; see tasks/StealQueue.hpp";
00150     doSteal = false; doShare = true; doGQ = false;
00151   } else if ( FLAGS_load_balance.compare( "gq" ) == 0 ) {
00152     CHECK( false ) << "--load_balance=gq currently unsupported; see tasks/StealQueue.hpp";
00153     doSteal = false; doShare = false; doGQ = true;
00154   } else {
00155     CHECK( false ) << "load_balance=" << FLAGS_load_balance << "; must be {none, steal, share, gq}";
00156   }
00157 
00158   fast_srand(0);
00159 
00160   localId = localId_arg;
00161   neighbors = neighbors_arg;
00162   numLocalNodes = numLocalNodes_arg;
00163   chunkSize = FLAGS_chunk_size;
00164 
00165   // initialize neighbors to steal permutation
00166   srandom(0);
00167   for (int i=numLocalNodes; i>=2; i--) {
00168     int ri = random() % i;
00169     Core temp = neighbors[ri];
00170     neighbors[ri] = neighbors[i-1];
00171     neighbors[i-1] = temp;
00172   }
00173 }
