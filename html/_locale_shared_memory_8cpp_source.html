00077 void LocaleSharedMemory::create() {
00078   VLOG(2) << "Creating LocaleSharedMemory region " << region_name 
00079           << " with " << region_size << " bytes"
00080           << " on " << global_communicator.mycore 
00081           << " of " << global_communicator.cores;
00082 
00083   // if possible, delete this user's old leftover share memory regions
00084   // if this returns false, either there was nothing to delete or
00085   // there was something owned by somebody else.
00086   bool deleted_one = boost::interprocess::shared_memory_object::remove( region_name.c_str() );
00087   if( deleted_one ) {
00088     VLOG(1) << "Deleted an old, leftover " << region_name.c_str();
00089   }
00090 
00091   // now, try to allocate a new one
00092   try {
00093     segment = boost::interprocess::fixed_managed_shared_memory( boost::interprocess::create_only,
00094                                                                 region_name.c_str(),
00095                                                                 region_size,
00096                                                                 base_address);
00097   }
00098   catch(...){
00099     LOG(ERROR) << "Failed to create locale shared memory of size " << region_size;
00100     boost::interprocess::shared_memory_object::remove( region_name.c_str() );
00101     failure_function();
00102     throw;
00103   }
00104   VLOG(2) << "Created LocaleSharedMemory region " << region_name 
00105           << " with " << region_size << " bytes"
00106           << " on " << global_communicator.mycore 
00107           << " of " << global_communicator.cores;
00108 }
