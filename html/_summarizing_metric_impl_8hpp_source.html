00045   void SummarizingMetric<T>::merge_all(impl::MetricBase* static_stat_ptr) {
00046     this->value_ = 0;
00047     this->n = 0;
00048     this->mean = 0.0;
00049     this->M2 = 0.0;
00050     this->max = 0;
00051     this->min = 0;
00052     
00053     // TODO: use more generalized `reduce` operation to merge all
00054     SummarizingMetric<T>* this_static = reinterpret_cast<SummarizingMetric<T>*>(static_stat_ptr);
00055     
00056     GlobalAddress<SummarizingMetric<T>> combined_addr = make_global(this);
00057     
00058     CompletionEvent ce(Grappa::cores());
00059     
00060     for (Core c = 0; c < Grappa::cores(); c++) {
00061       // we can compute the GlobalAddress here because we have pointers to globals,
00062       // which are guaranteed to be the same on all nodes
00063       GlobalAddress<SummarizingMetric<T>> remote_stat = make_global(this_static, c);
00064       
00065       send_heap_message(c, [remote_stat, combined_addr, &ce] {
00066           SummarizingMetric<T>* s = remote_stat.pointer();
00067           struct {
00068             T value_, min, max; size_t n; double mean, M2;
00069           } sp = {
00070             s->value_, s->min, s->max, s->n, s->mean, s->M2
00071           };
00072           
00073           send_heap_message(combined_addr.core(), [combined_addr, sp, &ce] {
00074               SummarizingMetric<T>* combined_ptr = combined_addr.pointer();
00075               if( sp.n > 0 ) { // update only if we have accumulated samples
00076                 combined_ptr->value_ += sp.value_;
00077                 if( 0 == combined_ptr->n ) { // overwrite if we are the first samples
00078                   combined_ptr->n = sp.n;
00079                   combined_ptr->mean = sp.mean;
00080                   combined_ptr->M2 = sp.M2;
00081                   combined_ptr->min = sp.min;
00082                   combined_ptr->max = sp.max;
00083                 } else {
00084                   size_t new_n = sp.n + combined_ptr->n;
00085                   double delta = sp.mean - combined_ptr->mean;
00086                   double new_mean = combined_ptr->mean + delta * ( static_cast<double>(sp.n) / new_n );
00087                   double new_M2 = combined_ptr->M2 + sp.M2 + 
00088                     delta * delta * combined_ptr->n * sp.n / new_n;
00089                   
00090                   combined_ptr->n = new_n;
00091                   combined_ptr->mean = new_mean;
00092                   combined_ptr->M2 = new_M2;
00093                   combined_ptr->max = std::max(combined_ptr->max, sp.max);
00094                   combined_ptr->min = std::min(combined_ptr->min, sp.min);
00095                 }
00096               }
00097               ce.complete();
00098             });
00099         });
00100     }
00101     ce.wait();
00102   }
