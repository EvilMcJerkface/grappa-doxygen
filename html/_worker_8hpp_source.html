00216 static void tramp(Worker * me, void * arg) {
00217   // Pass control back and forth a few times to get the info we need.
00218   Worker * master = (Worker *) arg;
00219   Worker * my_thr = (Worker *) coro_invoke(me, master, NULL);
00220   thread_func f = (thread_func) coro_invoke(me, master, NULL);
00221   void * f_arg = coro_invoke(me, master, NULL);
00222   // Next time we're invoked, it'll be for real.
00223   coro_invoke(me, master, NULL);
00224 
00225   StateTimer::setThreadState( StateTimer::SYSTEM );
00226   StateTimer::enterState_system();
00227   
00228   // create new Tau task, and top level timer for the task
00229 #ifdef GRAPPA_TRACE

00233   thread_last_tau_taskid = new_taskid;
00234   GRAPPA_PROFILE_CREATE( mainprof, "start_thread", "()", TAU_DEFAULT );
00235   GRAPPA_PROFILE_THREAD_START( mainprof, my_thr );
00236 #endif
00237 
00238   // call thread target function
00239   f(my_thr, f_arg);
00240 
00241   // stop top level Tau task timer
00242   GRAPPA_PROFILE_THREAD_STOP( mainprof, my_thr );

00244   // We shouldn't return, but if we do, kill the Worker.
