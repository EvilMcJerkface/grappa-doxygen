165 template< typename T, Core MASTER_CORE = 0 >
166 GlobalAddress<T> symmetric_global_alloc() {
167  static_assert(sizeof(T) % block_size == 0,
168  "must pad global proxy to multiple of block_size, or use GRAPPA_BLOCK_ALIGNED");
169  // allocate enough space that we are guaranteed to get one on each core at same location
170  auto qac = global_alloc<char>(cores()*(sizeof(T)+block_size));
171  while (qac.core() != MASTER_CORE) qac++;
172  auto qa = static_cast<GlobalAddress<T>>(qac);
173  CHECK_EQ(qa, qa.block_min());
174  CHECK_EQ(qa.core(), MASTER_CORE);
175  return qa;
176 }
