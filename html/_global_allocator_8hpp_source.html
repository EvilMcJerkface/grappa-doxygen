00165 template< typename T, Core MASTER_CORE = 0 >
00166 GlobalAddress<T> symmetric_global_alloc() {
00167   static_assert(sizeof(T) % block_size == 0,
00168                 "must pad global proxy to multiple of block_size, or use GRAPPA_BLOCK_ALIGNED");
00169   // allocate enough space that we are guaranteed to get one on each core at same location
00170   auto qac = global_alloc<char>(cores()*(sizeof(T)+block_size));
00171   while (qac.core() != MASTER_CORE) qac++;
00172   auto qa = static_cast<GlobalAddress<T>>(qac);
00173   CHECK_EQ(qa, qa.block_min());
00174   CHECK_EQ(qa.core(), MASTER_CORE);
00175   return qa;
00176 }
