55  // TODO: squeeze these all into one 64-bit word (the last word of the buffer)

57  // Buffer_ must be 64-byte aligned, so we only need 42 bits max of

89  std::tuple< void *, int > take_buffer() {
90  flush();
91 #ifdef USE_NT_OPS
92  _mm_sfence();
93 #endif
94  auto retval = std::make_tuple( reinterpret_cast<void*>(buffer_), position_ * sizeof(uint64_t) );
95  buffer_ = nullptr;
96  position_ = 0;
97  DCHECK_EQ( local_position_, 0 );
98  return retval;
99  }

101  int flush() {
102  DVLOG(5) << "Flushing " << this << " with position " << position_ << ", local_position " << local_position_;
103  if( local_position_ > 0 ) { // skip unless we have something to write
104  for( int i = local_position_; i < local_buffer_size; ++i ) {
105  localbuf_[i] = 0;
106  }
107 
108  if( !buffer_ ) {
109  new_buffer();
110  }
111 
112  // write out full cacheline
113  __m128i * src = reinterpret_cast<__m128i*>( &localbuf_[0] );
114  __m128i * dst = reinterpret_cast<__m128i*>( &buffer_[position_] );
115 #ifdef USE_NT_OPS
116  _mm_stream_si128( dst+0, *(src+0) );
117  _mm_stream_si128( dst+1, *(src+1) );
118  _mm_stream_si128( dst+2, *(src+2) );
119  _mm_stream_si128( dst+3, *(src+3) );
120 #else
121  _mm_store_si128( dst+0, *(src+0) );
122  _mm_store_si128( dst+1, *(src+1) );
123  _mm_store_si128( dst+2, *(src+2) );
124  _mm_store_si128( dst+3, *(src+3) );
125 #endif
126 
127  position_ += local_buffer_size; // advance to next cacheline of output buffer
128  local_position_ = 0;
129  }
130  return position_ * sizeof(uint64_t);
131  }

133  inline int actual_enqueue( uint64_t * word_p, int word_size ) {
134  // now copy in message
135  while( word_size > 0 ) {
136  localbuf_[ local_position_ ] = *word_p;
137  ++word_p;
138  --word_size;
139  ++local_position_;
140 
141  if( local_position_ == local_buffer_size ) {
142  flush();
143  }
144  }
145  DVLOG(5) << "After enqueue, position is " << position_ << " and local position is " << local_position_;
146  return (position_ + local_position_) * sizeof(uint64_t);
147  }

149  inline int enqueue( uint64_t * word_p, int word_size ) {
150  DVLOG(5) << "Enqueuing " << word_size << " words to " << this << " from " << word_p << ", position is " << position_ << " and local position is " << local_position_;
151 
152  // start with offset if necessary
153  if( (position_ == 0) && (local_position_ == 0) ) {
154  uint64_t offset[ initial_offset ];
155  actual_enqueue( &offset[0], initial_offset ); // add space for header
156  }
157  return actual_enqueue( word_p, word_size );
158  }

162 template< typename T >
163 static inline int nt_enqueue( NTBuffer * b, T * p, int size ) {
164 #ifdef USE_NT_OPS
165  _mm_prefetch( b, _MM_HINT_NTA );
166  _mm_prefetch( reinterpret_cast<char*>(b)+64, _MM_HINT_NTA );
167 #endif
168  DCHECK_EQ( reinterpret_cast<uint64_t>(p) % 8, 0 ) << "Pointer must be 8-byte aligned";
169  DCHECK_EQ( size % 8, 0 ) << "Size must be a multiple of 8";
170  return b->enqueue( reinterpret_cast<uint64_t*>(p), size/8 );
171 }
