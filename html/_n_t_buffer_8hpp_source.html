00101   int flush() {
00102     DVLOG(5) << "Flushing " << this << " with position " << position_ << ", local_position " << local_position_;
00103     if( local_position_ > 0 ) { // skip unless we have something to write
00104       for( int i = local_position_; i < local_buffer_size; ++i ) {
00105         localbuf_[i] = 0;
00106       }
00107       
00108       if( !buffer_ ) {
00109         new_buffer();
00110       }
00111 
00112       // write out full cacheline
00113       __m128i * src = reinterpret_cast<__m128i*>( &localbuf_[0] );
00114       __m128i * dst = reinterpret_cast<__m128i*>( &buffer_[position_] );
00115 #ifdef USE_NT_OPS
00116       _mm_stream_si128( dst+0, *(src+0) );
00117       _mm_stream_si128( dst+1, *(src+1) );
00118       _mm_stream_si128( dst+2, *(src+2) );
00119       _mm_stream_si128( dst+3, *(src+3) );
00120 #else
00121       _mm_store_si128( dst+0, *(src+0) );
00122       _mm_store_si128( dst+1, *(src+1) );
00123       _mm_store_si128( dst+2, *(src+2) );
00124       _mm_store_si128( dst+3, *(src+3) );
00125 #endif
00126 
00127       position_ += local_buffer_size; // advance to next cacheline of output buffer
00128       local_position_ = 0;
00129     }
00130     return position_ * sizeof(uint64_t);
00131   }

00133   inline int actual_enqueue( uint64_t * word_p, int word_size ) {
00134     // now copy in message
00135     while( word_size > 0 ) {
00136       localbuf_[ local_position_ ] = *word_p;
00137       ++word_p;
00138       --word_size;
00139       ++local_position_;
00140 
00141       if( local_position_ == local_buffer_size ) {

00144     }
00145     DVLOG(5) << "After enqueue, position is " << position_ << " and local position is " << local_position_;

00147   }

00156     }
00157     return actual_enqueue( word_p, word_size );
