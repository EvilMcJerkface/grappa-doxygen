00130   void global_heap_init(size_t init_size) {
00131     // by default, will allocate as much shared memory as it is
00132     // possible to evenly split among the processors on a node
00133     if (init_size != -1) {
00134       impl::global_memory_size_bytes = init_size;
00135       impl::global_bytes_per_core = init_size / cores();
00136       impl::global_bytes_per_locale = init_size / locales();
00137       return;
00138     }
00139     
00140     // Decide how much memory we should allocate for global shared heap.
00141     // this uses the locale shared size calculated in LocaleSharedMemory.cpp
00142     auto sz = static_cast<int64_t>(FLAGS_locale_shared_size * FLAGS_global_heap_fraction);
00143     
00144     int64_t nnode = global_communicator.locales;
00145     int64_t ppn = global_communicator.locale_cores;
00146     
00147     int64_t bytes_per_core = sz / ppn;
00148     // round down to page size so we don't ask for too much
00149     bytes_per_core &= ~( (1L << 12) - 1 );
00150     
00151     // be aware of hugepages
00152     // Each core should ask for a multiple of 1GB hugepages
00153     // and the whole node should ask for no more than the total pages available
00154     if ( FLAGS_global_memory_use_hugepages ) {
00155       int64_t pages_per_core = bytes_per_core / (1L << 30);
00156       int64_t new_bpp = pages_per_core * (1L << 30);
00157       if (new_bpp == 0) {
00158         MASTER_ONLY VLOG(1) << "Allocating 1GB per core anyway.";
00159         new_bpp = 1L << 30;
00160       }
00161       MASTER_ONLY VLOG_IF(1, bytes_per_core != new_bpp) << "With ppn=" << ppn << ", can only allocate "
00162       << pages_per_core*ppn << " / " << FLAGS_node_memsize / (1L << 30) << " 1GB huge pages per node";
00163       bytes_per_core = new_bpp;
00164     }
00165     
00166     int64_t bytes = nnode * ppn * bytes_per_core;
00167     int64_t bytes_per_node = ppn * bytes_per_core;
00168     MASTER_ONLY DVLOG(2) << "bpp = " << bytes_per_core << ", bytes = " << bytes << ", bytes_per_node = " << bytes_per_node
00169     << ", node_memsize = " << FLAGS_node_memsize << ", heap_size = " << sz;
00170     MASTER_ONLY VLOG(2) << "nnode: " << nnode << ", ppn: " << ppn << ", iBs/node: " << log2((double)bytes_per_node) << ", total_iBs: " << log2((double)bytes);
00171     
00172     impl::global_memory_size_bytes = bytes;
00173     impl::global_bytes_per_core = bytes_per_core;
00174     impl::global_bytes_per_locale = bytes_per_node;
00175   }

00186   StateTimer::setThreadState( StateTimer::COMMUNICATION );
00187   StateTimer::enterState_communication();
00188   while( !Grappa_done() ) {
00189     global_scheduler.stats.sample();
00190 
00191     Grappa::impl::poll();
00192     
00193     // check async. io completions
00194     if (aio_completed_stack) {
00195       // atomically grab the stack, replacing it with an empty stack again
00196       IODescriptor * desc = __sync_lock_test_and_set(&aio_completed_stack, NULL);
00197 
00198       while (desc != NULL) {
00199         desc->handle_completion();
00200         IODescriptor * temp = desc->nextCompleted;
00201         desc->nextCompleted = NULL;
00202         desc = temp;
00203       }
00204     }
00205 
00206     Grappa::yield_periodic();
00207   }
00208   // cleanup stragglers on readyQ since I should be last to run;
00209   // no one else matters.
00210   // Tasks on task queues would be a programmer error
00211   global_scheduler.shutdown_readyQ();
00212   VLOG(5) << "polling Worker exiting";
00213 
00214   // master will be scheduled upon exit of poller thread
00215 }

00287 void adjust_footprints() {
00288   auto locale_cores = global_communicator.locale_cores;
00289   auto locale_total = FLAGS_locale_shared_size;
00290   
00291   // (FLAGS_locale_shared_size either set manually or computed from fraction of node_memsize)
00292   auto locale_heap_bytes = static_cast<size_t>(FLAGS_locale_user_heap_fraction * (double)locale_total);
00293   auto global_heap_bytes = Grappa::impl::global_memory_size_bytes;
00294   
00295   // memory left for Grappa components
00296   auto grappa_bytes = (locale_total - global_heap_bytes - locale_heap_bytes) / locale_cores;
00297   CHECK_GT(grappa_bytes, 0)
00298     << "\nMust leave some memory for Grappa system components!\n"
00299     << " - locale_heap_bytes: " << locale_heap_bytes << "\n"
00300     << " - global_heap_bytes: " << global_heap_bytes << "\n"
00301     << " - total:    " << locale_total;
00302   
00303   auto total_footprint = []{
00304     return global_communicator.estimate_footprint()
00305     + global_rdma_aggregator.estimate_footprint()
00306     + global_task_manager.estimate_footprint()
00307     + SharedMessagePool::estimate_footprint();
00308   };
00309   
00310   if (total_footprint() < grappa_bytes) return;
00311   
00312   // otherwise try to get all the grappa components to play along
00313   long remaining = grappa_bytes;
00314   
00315   try {
00316     
00317     remaining -= SharedMessagePool::adjust_footprint(remaining / 4);
00318     if (remaining < 0) throw "SharedMessagePool";
00319     if (total_footprint() < grappa_bytes) throw true;
00320     
00321     remaining -= global_communicator.adjust_footprint(remaining / 3);
00322     if (remaining < 0) throw "Communicator";
00323     if (total_footprint() < grappa_bytes) throw true;
00324     
00325     remaining -= global_rdma_aggregator.adjust_footprint(remaining / 2);
00326     if (remaining < 0) throw "RDMA Aggregator";
00327     if (total_footprint() < grappa_bytes) throw true;
00328     
00329     remaining -= global_task_manager.adjust_footprint(remaining);
00330     if (remaining < 0) throw "TaskManager";
00331     if (total_footprint() < grappa_bytes) throw true;
00332     
00333   } catch (bool success) {
00334     
00335     MASTER_ONLY LOG(INFO) << "\nFootprint estimates: "
00336     << "\n- locale_heap_bytes: " << locale_heap_bytes
00337     << "\n- global_heap_bytes: " << global_heap_bytes
00338     << "\n- total for Grappa:  " << grappa_bytes
00339     << "\n  - global_communicator:    " << global_communicator.estimate_footprint()
00340     << "\n  - global_rdma_aggregator: " << global_rdma_aggregator.estimate_footprint()
00341     << "\n  - global_task_manager:    " << global_task_manager.estimate_footprint();
00342 
00343   } catch (char const* component) {
00344     MASTER_ONLY LOG(ERROR)
00345       << "\nUnable to fit Grappa components in memory. Failed at " << component
00346       << "\n  locale_heap_bytes:      " << locale_heap_bytes
00347       << "\n  global_heap_bytes:      " << global_heap_bytes
00348       << "\n  total for Grappa:       " << grappa_bytes
00349       << "\n  global_communicator:    " << global_communicator.estimate_footprint()
00350       << "\n  global_rdma_aggregator: " << global_rdma_aggregator.estimate_footprint()
00351       << "\n  global_task_manager:    " << global_task_manager.estimate_footprint()
00352       << "\n  shared_message_pool:    " << SharedMessagePool::estimate_footprint();
00353     exit(1);
00354   }

00602 void Grappa_activate() 
00603 {
00604   DVLOG(2) << "Activating Grappa library....";
00605   
00606   locale_shared_memory.activate(); // do this before communicator
00607   auto base_locale_shared_memory_allocated = locale_shared_memory.get_allocated();
00608 
00609   global_communicator.activate();
00610   auto communicator_locale_shared_memory_allocated = locale_shared_memory.get_allocated();
00611 
00612   global_task_manager.activate();
00613   auto tasks_locale_shared_memory_allocated = locale_shared_memory.get_allocated();
00614 
00615   global_communicator.barrier();
00616 
00617   // initializes system_wide global_memory pointer
00618   global_communicator.allreduce_inplace( &Grappa::impl::global_memory_size_bytes, MPI_INT64_T, MPI_MIN );
00619   global_memory = new GlobalMemory( Grappa::impl::global_memory_size_bytes );
00620   auto heap_locale_shared_memory_allocated = locale_shared_memory.get_allocated();
00621 
00622   // fire up polling thread
00623   global_scheduler.periodic( impl::worker_spawn( master_thread, &global_scheduler, &poller, NULL ) );
00624   auto polling_locale_shared_memory_allocated = locale_shared_memory.get_allocated();
00625 
00626   global_rdma_aggregator.activate();
00627   auto aggregator_locale_shared_memory_allocated = locale_shared_memory.get_allocated();
00628   
00629   SharedMessagePool::activate();
00630   auto shared_pool_locale_shared_memory_allocated = locale_shared_memory.get_allocated();
00631   
00632   if (Grappa::mycore() == 0) {
00633     double node_sz_gb = static_cast<double>(FLAGS_node_memsize) / (1L<<30);
00634     double locale_sz_gb = static_cast<double>(FLAGS_locale_shared_size) / (1L<<30);
00635     double locale_core_sz_gb = static_cast<double>(FLAGS_locale_shared_size) / Grappa::locale_cores() / (1L<<30);
00636     double communicator_sz_gb = static_cast<double>( communicator_locale_shared_memory_allocated - base_locale_shared_memory_allocated ) / (1L<<30);
00637     double tasks_sz_gb = static_cast<double>( tasks_locale_shared_memory_allocated - communicator_locale_shared_memory_allocated ) / (1L<<30);
00638     double heap_sz_gb = static_cast<double>( heap_locale_shared_memory_allocated - tasks_locale_shared_memory_allocated ) / (1L<<30);
00639     tasks_sz_gb += static_cast<double>( polling_locale_shared_memory_allocated - heap_locale_shared_memory_allocated ) / (1L<<30);
00640     double aggregator_sz_gb = static_cast<double>( aggregator_locale_shared_memory_allocated - polling_locale_shared_memory_allocated ) / (1L<<30);
00641     double shared_pool_sz_gb = static_cast<double>( shared_pool_locale_shared_memory_allocated - aggregator_locale_shared_memory_allocated ) / (1L<<30);
00642     double shared_pool_max_sz_gb = static_cast<double>( FLAGS_shared_pool_max_size ) / (1L<<30);
00643     
00644     size_t free_sz = static_cast<double>(Grappa::impl::locale_shared_memory.get_free_memory());
00645     double free_sz_gb = static_cast<double>(free_sz) / (1L<<30);
00646     double free_core_sz_gb = static_cast<double>(free_sz) / Grappa::locale_cores() / (1L<<30);
00647     VLOG(1) << "\n-------------------------\nShared memory breakdown:\n"
00648             << "  node total:                   " << node_sz_gb << " GB\n"
00649             << "  locale shared heap total:     " << locale_sz_gb << " GB\n"
00650             << "  locale shared heap per core:  " << locale_core_sz_gb << " GB\n"
00651             << "  communicator per core:        " << communicator_sz_gb << " GB\n"
00652             << "  tasks per core:               " << tasks_sz_gb << " GB\n"
00653             << "  global heap per core:         " << heap_sz_gb << " GB\n"
00654             << "  aggregator per core:          " << aggregator_sz_gb << " GB\n"
00655             << "  shared_pool current per core: " << shared_pool_sz_gb << " GB\n"
00656             << "  shared_pool max per core:     " << shared_pool_max_sz_gb << " GB\n"
00657             << "  free per locale:              " << free_sz_gb << " GB\n"
00658             << "  free per core:                " << free_core_sz_gb << " GB\n"
00659             << "-------------------------";
00660 
00661     CHECK_GT( free_core_sz_gb, shared_pool_max_sz_gb ) 
00662       << "Not enough free locale shared heap for fully-allocated shared message pool";
00663   }
00664   
00665   global_communicator.barrier();
00666 }
