158  Core origin = mycore();

160 
161  auto lsz = [&ce,origin,work]{};
162 
163  for (Core c = 0; c < cores(); c++) if (c != mycore()) {
164  send_heap_message(c, [&ce, origin, work] {

212  template< typename T, T (*ReduceOp)(const T&, const T&) >
213  void collect_reduction(const T& val) {
214  static T total;
215  static Core cores_in = 0;
216 
217  DCHECK(mycore() == HOME_CORE);
218 
219  if (cores_in == 0) {
220  total = val;
221  } else {
222  total = ReduceOp(total, val);
223  }
224 
225  cores_in++;
226  DVLOG(4) << "cores_in: " << cores_in;
227 
228  if (cores_in == cores()) {
229  cores_in = 0;
230  T tmp_total = total;
231  for (Core c = 0; c < cores(); c++) {
232  send_heap_message(c, [tmp_total] {
233  Reduction<T>::result.writeXF(tmp_total);
234  });
235  }
236  }

239  template<typename T, T (*ReduceOp)(const T&, const T&) >
240  class InplaceReduction {
241  protected:
242  CompletionEvent * ce;
243  T * array;
244  Core elems_in = 0;
245  size_t nelem;
246  public:
249  void call_allreduce(T * in_array, size_t nelem) {
250  // setup everything (block to make sure HOME_CORE is done)
251  this->array = in_array;
252  this->nelem = nelem;
253 
254  size_t n_per_msg = MAX_MESSAGE_SIZE / sizeof(T);
255  size_t nmsg = nelem / n_per_msg + (nelem % n_per_msg ? 1 : 0);
256  auto nmsg_total = nmsg*(cores()-1);
257 
258  CompletionEvent local_ce;
259  this->ce = &local_ce;
260  this->ce->enroll( (mycore() == HOME_CORE) ? nmsg_total : nmsg );
261  barrier();
262 
263  if (mycore() != HOME_CORE) {
264  for (size_t k=0; k<nelem; k+=n_per_msg) {
265  size_t this_nelem = std::min(n_per_msg, nelem-k);
266 
267  // everyone sends their contribution to HOME_CORE, last one wakes HOME_CORE
268  send_heap_message(HOME_CORE, [this,k](void * payload, size_t payload_size) {
269  DCHECK(mycore() == HOME_CORE);
270 
271  auto in_array = static_cast<T*>(payload);
272  auto in_n = payload_size/sizeof(T);
273  auto total = this->array+k;
274 
275  for (size_t i=0; i<in_n; i++) {
276  total[i] = ReduceOp(total[i], in_array[i]);
277  }
278  DVLOG(3) << "incrementing HOME sem, now at " << ce->get_count();
279  this->ce->complete();
280  }, (void*)(in_array+k), sizeof(T)*this_nelem);
281  }
282 
283  DVLOG(3) << "about to block for " << nelem << " with sem == " << ce->get_count(); this->ce->wait();
284 
285  } else {
286  auto nmsg_total = nmsg*(cores()-1);
287 
288  // home core waits until woken by last received message from other cores
289  this->ce->wait();
290  DVLOG(3) << "woke with sem == " << ce->get_count();
291 
292  // send total to everyone else and wake them
293  char msg_buf[(cores()-1)*sizeof(PayloadMessage<std::function<void(decltype(this),size_t)>>)];
294  MessagePool pool(msg_buf, sizeof(msg_buf));
295  for (Core c = 0; c < cores(); c++) {
296  if (c != HOME_CORE) {
297  // send totals back to all the other cores
298  size_t n_per_msg = MAX_MESSAGE_SIZE / sizeof(T);
299  for (size_t k=0; k<nelem; k+=n_per_msg) {
300  size_t this_nelem = std::min(n_per_msg, nelem-k);
301  pool.send_message(c, [this,k](void * payload, size_t psz){
302  auto total_k = static_cast<T*>(payload);
303  auto in_n = psz / sizeof(T);
304  for (size_t i=0; i<in_n; i++) {
305  this->array[k+i] = total_k[i];
306  }
307  this->ce->complete();
308  DVLOG(3) << "incrementing sem, now at " << ce->get_count();
309  }, this->array+k, sizeof(T)*this_nelem);
310  }
311  }
312  }
313  // once all messages are sent, HOME_CORE's task continues
314  }
315  }
316  };

368  template< typename T, T (*ReduceOp)(const T&, const T&) >
369  T reduce(const T * global_ptr) {
370  //NOTE: this is written in a continuation passing
371  //style to avoid the use of a GCE which async delegates only support
372  CompletionEvent ce(cores()-1);
373 
374  T total = *global_ptr;
375  Core origin = mycore();
376 
377  for (Core c=0; c<cores(); c++) {
378  if (c != origin) {
379  send_heap_message(c, [global_ptr, &ce, &total, origin]{
380  T val = *global_ptr;
381  send_heap_message(origin, [val,&ce,&total] {
382  total = ReduceOp(total, val);
383  ce.complete();
384  });
385  });
386  }
387  }
388  ce.wait();
389  return total;
390  }
