00096         static_assert(std::is_same<R, decltype(func())>::value, "return type of callable must match the type of this Promise");

00098         delegate_ops++;
00099         delegate_async_ops++;

00101         
00102         if (dest == origin) {

00104           delegate_targets++;
00105           delegate_short_circuits++;
00106           fill(func());

00108           start_time = Grappa::timestamp();
00109

00111             delegate_targets++;
